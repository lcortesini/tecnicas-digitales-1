\section{Ejercicio 1}

$$
F(x,y,z) = \sum(2,3,4,5)
$$

\subsection{Descripción en Verilog}

\begin{Verbatim}[ frame=single, fontsize=\small, baselinestretch=1.2 ]
module ej1 (
    input x,
    input y,
    input z,
    output F
);

endmodule
\end{Verbatim}
